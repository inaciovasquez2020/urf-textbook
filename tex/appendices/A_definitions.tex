\chapter{Core Definitions}

\begin{definition}[Entropy]
Entropy \(H(X)\) of a distribution \(X\) is

\[
H(X) = -\sum_x p(x)\log p(x).
\]
\end{definition}

\begin{definition}[Capacity]
Capacity \(C\) denotes the maximal admissible information throughput of the system.
\end{definition}

\begin{definition}[EntropyDepth]
EntropyDepth \(D\) is the minimal number of refinement steps required to collapse configuration entropy.
\end{definition}
