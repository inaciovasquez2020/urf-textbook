\section{Open Problems on $\chi_R$}

\begin{problem}[Growth of $\chi_R$]
Characterize bounded-degree families with $\chi_R(G_n)=\omega(1)$.
\end{problem}

\begin{problem}[Optimal Scheduling]
Given $G_n$, determine minimal rounds under capacity $c$.
\end{problem}

\begin{problem}[Randomized Refinement]
Determine expected rounds for random $R$-independent activation.
\end{problem}
