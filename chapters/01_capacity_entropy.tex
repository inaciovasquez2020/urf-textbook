\chapter{Capacity and Entropy}

\section{Definitions}

\begin{definition}[Information Capacity]
Information capacity $C$ is the maximal extractable information under admissible refinement.
\end{definition}

\section{Core Inequality}

\begin{theorem}[CEICI]
For any admissible refinement system,
\[
\Phi(x) = H(x) - C(x) \ge 0.
\]
\end{theorem}
