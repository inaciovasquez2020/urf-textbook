\section{Packing Bound}

\begin{lemma}[Packing Bound]
Let $G_n$ have maximum degree $\Delta$. For fixed $R\ge1$,
\[
\alpha_R(G_n) \ge \frac{n}{B_\Delta(2R)},
\]
where
\[
B_\Delta(2R)=1+\Delta\sum_{i=0}^{2R-1}(\Delta-1)^i.
\]
\end{lemma}

\begin{proof}
A greedy packing of radius-$2R$ balls covers at most $B_\Delta(2R)$ vertices per center.
\end{proof}
