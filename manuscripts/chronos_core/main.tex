\begin{abstract}
We formalize the Chronos refinement framework as a locality-constrained
information-bandwidth process.
We prove two structural rigidity results:
(1) bounded-degree locality imposes a linear entropy collapse ceiling,
and (2) extraction of global invariants requires linear EntropyDepth.
The arguments are purely combinatorial and rely on finite
neighborhood-type enumeration.
\end{abstract}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Chronos Refinement Model}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Let $G=(V,E)$ be a finite graph with maximum degree $\Delta$.
Let $n = |V|$.

For $r \ge 0$, define:
\[
\Ball_r(x) := \{ y \in V : \operatorname{dist}(x,y) \le r \}.
\]

Let
\[
\tau_t : V \to C_t
\]
be refinement colorings.

Let
\[
Q_t : C_t \to D_t
\]
be coarsening maps.

\begin{definition}[Transcript Entropy]
\[
H_t := \log |D_t|.
\]
\end{definition}

\begin{definition}[EntropyDepth]
\[
\ED(G) := \min \{ t : |D_t| = 1 \}.
\]
\end{definition}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Faithful Local Coarsening (FLC)}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{definition}[FLC$(\Delta,r,k)$]
The system satisfies:

\begin{enumerate}
\item \textbf{Local Determination:}
\[
Q_{t+1}(\tau_{t+1}(x))
=
F_t\!\left(\{\!\{\tau_t(y):y\in\Ball_r(x)\}\!\}\right).
\]

\item \textbf{$\FO^k$-Faithfulness:}
If $Q_t(\tau_t(x)) = Q_t(\tau_t(x'))$,
then $x,x'$ satisfy the same $\FO^k$ formulas
of quantifier rank $\le r$.

\item \textbf{Monotone Factoring:}
\[
|D_{t+1}| \le |D_t|.
\]
\end{enumerate}
\end{definition}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Finite Neighborhood Type Space}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Let
\[
\Types_{k,r,\Delta}
\]
denote the set of $\FO^k$-types of rooted radius-$r$ balls
in graphs of degree $\le \Delta$.

\begin{lemma}[Finite Type Bound]
For fixed $(\Delta,r,k)$,
\[
C(\Delta,r,k)
:=
|\Types_{k,r,\Delta}|
< \infty.
\]
\end{lemma}

This follows from classical finite model theory.

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Bounded-Degree Local Collapse Ceiling}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{lemma}[Local Entropy Ceiling]
Under FLC$(\Delta,r,k)$,
\[
H_t - H_{t+1}
\le
\log C(\Delta,r,k).
\]
\end{lemma}

\begin{proof}
By locality, $Q_{t+1}$ depends only on radius-$r$ neighborhoods.
By faithfulness, collapse occurs only within $\FO^k$-types.
There are at most $C(\Delta,r,k)$ such types.
Hence
\[
|D_{t+1}|
\ge
\frac{|D_t|}{C(\Delta,r,k)},
\]
which yields the bound.
\end{proof}

\begin{theorem}[Bounded-Degree Linear Collapse]
If $\Delta$ and $r$ are fixed,
\[
H_0 - H_t
\le
t \cdot \log C(\Delta,r,k).
\]
\end{theorem}

\begin{proof}
Sum the per-step bound.
\end{proof}

\begin{corollary}
Under fixed $\Delta,r$,
\[
H_0 - H_t = \omega(t)
\]
is impossible.
\end{corollary}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Global Amplification Barrier}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{definition}[Global Invariant]
A function $\Phi$ on graphs is global if
for every fixed $r$ there exist graphs $G,H$
with isomorphic radius-$r$ neighborhoods
but $\Phi(G) \ne \Phi(H)$.
\end{definition}

\begin{theorem}[Global Amplification Barrier]
Let $\Delta,r$ be fixed.
Assume FLC$(\Delta,r,k)$.
Let $\Phi$ be a global invariant.

Then any refinement process computing $\Phi$
requires
\[
\ED(G) = \Omega(\log n)
\]
for infinitely many $n$.
\end{theorem}

\begin{proof}
A global invariant is not determined by bounded-radius neighborhoods.
Thus no bounded number of locality-restricted coarsening steps
can collapse all indistinguishable graphs into $\Phi$-classes.
Each step reduces entropy by at most $\log C(\Delta,r,k)$.
Since initial entropy is $\Theta(\log n)$,
at least $\Omega(\log n)$ steps are required.
\end{proof}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\section{Structural Interpretation}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

Chronos establishes two rigidity walls:

\begin{itemize}
\item Bounded-degree locality imposes finite per-step bandwidth.
\item Global invariant extraction requires linear EntropyDepth.
\end{itemize}

Thus local refinement cannot amplify bounded information
into global structure without paying linear information work.
