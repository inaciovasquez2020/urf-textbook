\begin{lemma}[Sequential Information Bound]
\textbf{Dependencies: } D1, L1
\textbf{Status: } Proven

Let $(R_1,\dots,R_T)$ be a refinement sequence with transcript variables
$T_1,\dots,T_T$.

Then total information satisfies
\[
I(X;T_1,\dots,T_T)
= \sum_{t=1}^T I(X;T_t \mid T_1,\dots,T_{t-1})
\leq \sum_{t=1}^T C_t,
\]
where $C_t$ denotes the capacity bound of step $t$.

Thus information accumulation is additive and capacity-limited.
\end{lemma}
