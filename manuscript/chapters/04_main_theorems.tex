\chapter{Main Theorems}
\begin{theorem}[EntropyDepth Lower Bound]
For transcript-limited refinement systems on instances with $H=\Theta(n)$, the EntropyDepth satisfies $\mathrm{ED}\ge \Omega(n)$.
\end{theorem}
\begin{theorem}[Terminal Obstruction Template]
Let $L$ be an operator and $V$ a defect subspace. If $\lambda_1(L|V^\perp)>0$ then coercive closure holds in the corresponding reduction.
\end{theorem}
