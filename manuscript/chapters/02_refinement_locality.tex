\chapter{Refinement and Locality}
\begin{definition}[Refinement System]
A refinement system is a state space $\mathcal S$ with maps $R_t:\mathcal S\to\mathcal S$.
\end{definition}
\begin{definition}[$\mathrm{FO}^k$-Local Refinement]
A refinement is $\mathrm{FO}^k$-local if its distinguishing power is bounded by $\mathrm{FO}^k$ types at fixed radius.
\end{definition}
