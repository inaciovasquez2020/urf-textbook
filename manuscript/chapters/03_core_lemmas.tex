\chapter{Core Lemmas}
\begin{lemma}[Non-Amplification]
In a transcript-limited refinement system, net extractable information does not exceed transcript capacity.
\end{lemma}
\begin{lemma}[Width--Entropy Bridge]
An entropy drop $\delta$ implies width $\Omega(\log(1/\delta))$ under bounded-local reasoning.
\end{lemma}
